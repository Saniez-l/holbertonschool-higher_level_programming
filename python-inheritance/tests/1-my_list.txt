Test the MyList class and its print_sorted method.

>>> MyList = __import__('1-my_list').MyList

>>> my_list = MyList()
>>> my_list.append(1)
>>> my_list.append(4)
>>> my_list.append(2)
>>> my_list.append(3)
>>> my_list.append(5)
>>> print(my_list)
[1, 4, 2, 3, 5]
>>> my_list.print_sorted()
[1, 2, 3, 4, 5]
>>> print(my_list)
[1, 4, 2, 3, 5]

>>> empty_list = MyList()
>>> print(empty_list)
[]
>>> empty_list.print_sorted()
[]

>>> sorted_list = MyList([1, 2, 3, 4, 5])
>>> print(sorted_list)
[1, 2, 3, 4, 5]
>>> sorted_list.print_sorted()
[1, 2, 3, 4, 5]

>>> reverse_list = MyList([5, 4, 3, 2, 1])
>>> print(reverse_list)
[5, 4, 3, 2, 1]
>>> reverse_list.print_sorted()
[1, 2, 3, 4, 5]
>>> print(reverse_list)
[5, 4, 3, 2, 1]
